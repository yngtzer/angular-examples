  • A function and 'set' keyword appeared after @Input() means that whenever the value passing in changes, the function will be executed.
  • TemplateRef - To instantiate the embedded view declare within the opening and closing tag
  • ViewContainerRef - Location to place the embedded view
        
          constructor(private templateRef: TemplateRef<any>, private vcRef: ViewContainerRef) { }

          @Input() set appUnless(condition: boolean) {
            if (!condition) {
              this.vcRef.createEmbeddedView(this.templateRef);
            } else {
              this.vcRef.clear();
            }
          }
